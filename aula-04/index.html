         import random

def gerar_equacao_segundo_grau():
    # O coeficiente 'a' não pode ser 0 em uma equação do 2º grau
    a = random.randint(-10, 10)
    while a == 0:
        a = random.randint(-10, 10)
        
    b = random.randint(-20, 20)
    c = random.randint(-20, 20)

    # Função auxiliar para formatar a exibição dos termos
    def formatar_termo(valor, termo, primeiro=False):
        if valor == 0: return ""
        sinal = "+" if valor > 0 and not primeiro else ""
        
        # Ajusta exibição de coeficientes 1 e -1 (ex: x² em vez de 1x²)
        if valor == 1 and termo != "": valor_str = ""
        elif valor == -1 and termo != "": valor_str = "-"
        else: valor_str = str(valor)
        
        return f"{sinal}{valor_str}{termo}"

    # Montagem da string da equação ax² + bx + c = 0
    equacao = (formatar_termo(a, "x²", True) + 
               formatar_termo(b, "x") + 
               formatar_termo(c, ""))
    
    return f"{equacao} = 0"

# Exemplo de uso
nova_equacao = gerar_equacao_segundo_grau()
print(f"Equação gerada: {nova_equacao}")
